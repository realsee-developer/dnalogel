                strategy?: "FoldWhenMove";
            };
        disableFold?: true;
        disableUnfold?: true;
        keep?: "folded" | "unfolded";
        unfoldDistance?: MinMax;
    };
    visibleConfig?: {
        alwaysShowWhenMovePano?: boolean;
        angleRange?: MinMax;
        entryFromModel?: boolean;
        followModelVisibility?: boolean;

            distanceAccuracy?: number;
            enabled?: boolean;

        };
        keep?: "hidden" | "visible";
